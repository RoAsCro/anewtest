public class CalculatorTwo {

	double add(int a, int b) {
		return a + b;
	}

	double subtract(int a, int b) {
		return a - b;
	}

	double multiply(int a, int b) {
		return a * b;
	}

	double divide(int a, int b) {
		return (double) a / (double) b;
	}

	double modulus(int a, int b) {
		return (double) a % (double) b;
	}
}
