public class Item {
	private int x, y, z;
	private String name;
	private Item nextItem = null;

	public Item(int x, int y, int z, String name) {
		this.x = x;
		this.y = y;
		this.z = z;
		this.name = name;
	}

	public void setNextItem(Item item) {
		nextItem = item;
	}

	public Item getNextItem() {
		return nextItem;
	}
}
